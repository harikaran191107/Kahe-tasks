24)To print the second minutes and hours in time format:

#include<stdio.h>
int main(){
    int secs,min,hrs;
    printf("enter the hours:");
    scanf("%d",&hrs);
    printf("enter the minutes:");
    scanf("%d",&min);
    printf("enter the seconds:");
    scanf("%d",&secs);
    printf("now the time is %d:%d:%d",hrs,min,secs);
    return 0;
}
