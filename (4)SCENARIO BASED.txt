1)To print the price after discount:

#include<stdio.h>
int main(){
    int discount,totalamt,finalamt;
    printf("'Before discount'");
    printf("\n");
    printf("enter the amount:");
    scanf("%d",&totalamt);
    printf("enter the discount:");
    scanf("%d%",&discount);
    finalamt=(totalamt/100)*discount;
    printf("'After discount'");
    printf("\n");
    printf("the final amount is %d",finalamt);
    return 0;
}
